create table if not exists precios_diarios ( id uuid default gen_random_uuid() primary key, fecha date not null, -- Precios de compra Repsol (iguales para todas las estaciones) goa_pc numeric, g95_pc numeric, g98_pc numeric, nexa_pc numeric, -- PVP Grupo A (Europolis, Las Matas, El Escorial, Saltos del Sil, Monte del Pilar) goa_pvp_a numeric, g95_pvp_a numeric, g98_pvp_a numeric, nexa_pvp_a numeric, -- PVP Grupo B (Valdemoro, Arroyomolinos) goa_pvp_b numeric, g95_pvp_b numeric, g98_pvp_b numeric, nexa_pvp_b numeric, usuario_id uuid references auth.users(id), created_at timestamp default now() ); alter table precios_diarios enable row level security; create policy "Usuarios pueden insertar precios" on precios_diarios for insert to authenticated with check (true); create policy "Usuarios pueden leer precios" on precios_diarios for select to authenticated using (true); create policy "Usuarios pueden actualizar precios" on precios_diarios for update to authenticated using (true);
